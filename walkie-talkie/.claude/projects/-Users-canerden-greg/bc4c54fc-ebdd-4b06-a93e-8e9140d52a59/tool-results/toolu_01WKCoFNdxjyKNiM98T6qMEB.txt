   799→function calcTargets() {
   800→  const w=parseFloat(ob.weight)||70, h=parseFloat(ob.height)||170;
   801→  const bmr = ob.sex==='female' ? 10*w+6.25*h-5*25-161 : 10*w+6.25*h-5*25+5;
   802→  const tdee = Math.round(bmr*1.55);
   803→  let calories, protein;
   804→  if(ob.goal==='bulk')     { calories=tdee+300; protein=Math.round(w*2.0); }
   805→  else if(ob.goal==='cut') { calories=tdee-400; protein=Math.round(w*2.2); }
   806→  else                     { calories=tdee;     protein=Math.round(w*1.8); }
   807→  return { calories: Math.round(calories/50)*50, protein };

  1056→function matchExercise(text, aliases) {

  1064→function parseSets(text) {
  1065→  // Patterns: "80kg x8 x7 x8", "3x10 @70kg", "70 8 8 8", "80/82/85 x6"
  1066→  const sets=[];
  1067→  // pattern: weight×reps pairs like "80kg x8" repeated
  1068→  const pairPattern=/(\d+(?:\.\d+)?)\s*(?:kg)?\s*[x×]\s*(\d+)/gi;
  1069→  let m;
  1070→  while((m=pairPattern.exec(text))!==null){
  1071→    sets.push({weight:m[1],reps:m[2]});
  1072→  }
  1073→  if(sets.length) return sets;
  1074→  // pattern: "3x10 @70" or "3x10 70kg"
  1075→  const setsRepsWeight=/(\d+)\s*[x×]\s*(\d+)\s*@?\s*(\d+(?:\.\d+)?)/i.exec(text);
  1076→  if(setsRepsWeight){
  1077→    const n=parseInt(setsRepsWeight[1]);
  1078→    for(let i=0;i<n;i++) sets.push({weight:setsRepsWeight[3],reps:setsRepsWeight[2]});
  1079→    return sets;
  1080→  }
  1081→  // pattern: "weight x reps x reps" like "80 x 8 8 7"
  1082→  const wFirst=/(\d+(?:\.\d+)?)\s*(?:kg)?\s*[x×]\s*([\d\s]+)/i.exec(text);
  1083→  if(wFirst){
  1084→    const reps=wFirst[2].trim().split(/\s+/);
  1085→    reps.forEach(r=>{ if(/^\d+$/.test(r)) sets.push({weight:wFirst[1],reps:r}); });
  1086→    if(sets.length) return sets;
  1087→  }
  1088→  return [];
  1089→}

  1091→function parsePaste() {
  1092→  const text=document.getElementById('paste-input').value.trim();
  1093→  if(!text){alert('Paste some notes first.');return;}
  1094→  const aliases=buildAliases();
  1095→  const lines=text.split('\n').map(l=>l.trim()).filter(Boolean);
  1096→  const result={
  1097→    date: todayKey(),
  1098→    workoutKey: null,
  1099→    workoutName: null,
  1100→    exercises: [],
  1101→    weight: null,
  1102→    bf: null,
  1103→    habits: {},
  1104→  };
  1105→  const HABITS=getHabits();
  1106→
  1107→  for(const line of lines){
  1108→    const lower=line.toLowerCase();
  1109→
  1110→    // Date
  1111→    const dateM=line.match(/\b(20\d\d[-\/]\d{1,2}[-\/]\d{1,2})\b/);
  1112→    if(dateM){result.date=dateM[1].replace(/\//g,'-');continue;}
  1113→    const monthNames=['january','february','march','april','may','june','july','august','september','october','november','december'];
  1114→    for(let mi=0;mi<monthNames.length;mi++){
  1115→      const mn=new RegExp(`${monthNames[mi]}\\s+(\\d{1,2})(?:st|nd|rd|th)?(?:,?\\s*(\\d{4}))?`,'i').exec(line);
  1116→      if(mn){
  1117→        const y=mn[2]||new Date().getFullYear();
  1118→        const mo=String(mi+1).padStart(2,'0');
  1119→        const d=String(mn[1]).padStart(2,'0');
  1120→        result.date=`${y}-${mo}-${d}`;
  1121→      }
  1122→    }
  1123→
  1124→    // Workout type
  1125→    if(!result.workoutKey){
  1126→      for(const w of plan.workouts){
  1127→        const key=w.key.toLowerCase().replace(/[12]$/,'');
  1128→        if(lower.includes(`workout ${key}`)||lower===key||lower===`workout ${key}`||lower.startsWith(key+' ')||lower===w.name.toLowerCase()){
  1129→          result.workoutKey=w.key; result.workoutName=w.name; break;
  1130→        }
  1131→      }
  1132→    }
  1133→
  1134→    // Weight / BF
  1135→    const weightM=line.match(/(?:weight|bw|weighed?)[:\s]+(\d+(?:\.\d+)?)\s*(?:kg)?/i);
  1136→    if(weightM){result.weight=parseFloat(weightM[1]);continue;}
  1137→    const bfM=line.match(/(?:bf|bodyfat|body\s*fat)[:\s]+(\d+(?:\.\d+)?)\s*%?/i);
  1138→    if(bfM){result.bf=parseFloat(bfM[1]);continue;}
  1139→
  1140→    // Habits
  1141→    const habitPatterns={
  1142→      creatine:/creatine/i,
  1143→      shake:/shake|meal\s*prep|morning\s*meal/i,
